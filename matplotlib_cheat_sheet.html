# Create sample DataFrame import pandas as pd data = { 'date': pd.date_range('2024-01-01', periods=100, freq='D'), 'sales': np.random.normal(1000, 200, 100).cumsum(), 'costs': np.random.normal(800, 150, 100).cumsum(), 'region': np.random.choice(['North', 'South', 'East', 'West'], 100) } df = pd.DataFrame(data) df['profit'] = df['sales'] - df['costs'] # Create multi-panel visualization fig, axes = plt.subplots(2, 2, figsize=(15, 10)) # Time series plot df.set_index('date')[['sales', 'costs', 'profit']].plot(ax=axes[0,0]) axes[0,0].set_title('Financial Trends Over Time') axes[0,0].legend() # Regional sales distribution df.groupby('region')['sales'].sum().plot(kind='bar', ax=axes[0,1]) axes[0,1].set_title('Sales by Region') axes[0,1].tick_params(axis='x', rotation=45) # Profit vs Sales scatter df.plot.scatter(x='sales', y='profit', alpha=0.6, ax=axes[1,0]) axes[1,0].set_title('Profit vs Sales Relationship') # Profit distribution df['profit'].plot(kind='hist', bins=20, alpha=0.7, ax=axes[1,1]) axes[1,1].set_title('Profit Distribution') plt.tight_layout() plt.show()

3D Visualization
# Create 3D surface plot from mpl_toolkits.mplot3d import Axes3D # Generate data x = np.linspace(-5, 5, 50) y = np.linspace(-5, 5, 50) X, Y = np.meshgrid(x, y) Z = np.sin(np.sqrt(X**2 + Y**2)) # Create 3D plot fig = plt.figure(figsize=(12, 5)) # Surface plot ax1 = fig.add_subplot(121, projection='3d') surf = ax1.plot_surface(X, Y, Z, cmap='viridis', alpha=0.9) ax1.set_title('3D Surface Plot') ax1.set_xlabel('X') ax1.set_ylabel('Y') ax1.set_zlabel('Z') fig.colorbar(surf, shrink=0.5) # Wireframe plot ax2 = fig.add_subplot(122, projection='3d') ax2.plot_wireframe(X, Y, Z, color='black', alpha=0.6) ax2.set_title('3D Wireframe Plot') ax2.set_xlabel('X') ax2.set_ylabel('Y') ax2.set_zlabel('Z') plt.tight_layout() plt.show()

Performance Tips
plt.ioff() Turn off interactive mode
plt.ion() Turn on interactive mode
plt.clf() Clear current figure
plt.close('all') Close all figures
Performance Tip: Use plt.ioff() for batch processing. Always close figures with plt.close() to free memory. Use vector formats (PDF, SVG) for publications.
Memory Warning: Large datasets and many subplots can consume significant memory. Close figures when done and use appropriate data sampling.
