Principal Component Analysis
# PCA using SciPy linear algebra from scipy import linalg # Generate correlated data np.random.seed(42) mean = [0, 0] cov = [[1, 0.8], [0.8, 1]] data = np.random.multivariate_normal(mean, cov, 200) # Center the data data_centered = data - np.mean(data, axis=0) # Compute covariance matrix cov_matrix = np.cov(data_centered.T) # Eigenvalue decomposition eigenvals, eigenvecs = linalg.eigh(cov_matrix) # Sort by eigenvalues (descending) idx = np.argsort(eigenvals)[::-1] eigenvals = eigenvals[idx] eigenvecs = eigenvecs[:, idx] print(f"Eigenvalues: {eigenvals}") print(f"Explained variance: {eigenvals / np.sum(eigenvals)}") # Transform to principal components pca_data = data_centered @ eigenvecs

# Signal filtering example from scipy import signal from scipy.fft import fft, fftfreq # Generate noisy signal fs = 1000 # Sample rate t = np.linspace(0, 2, 2*fs, endpoint=False) clean_signal = (np.sin(2*np.pi*50*t) + 0.5*np.sin(2*np.pi*120*t)) noise = 0.2 * np.random.randn(len(t)) noisy_signal = clean_signal + noise # Design low-pass filter nyquist = fs / 2 low_cutoff = 100 / nyquist b, a = signal.butter(4, low_cutoff, btype='low') # Apply filter filtered_signal = signal.filtfilt(b, a, noisy_signal) # Find peaks in filtered signal peaks, _ = signal.find_peaks(filtered_signal, height=0.5) print(f"Found {len(peaks)} peaks in filtered signal")

# Solve system of ODEs (harmonic oscillator) from scipy import integrate def harmonic_oscillator(t, state): x, v = state dxdt = v dvdt = -x # Simple harmonic oscillator return [dxdt, dvdt] # Initial conditions initial_state = [1.0, 0.0] # x=1, v=0 t_span = (0, 10) t_eval = np.linspace(0, 10, 1000) # Solve ODE solution = integrate.solve_ivp( harmonic_oscillator, t_span, initial_state, t_eval=t_eval, method='RK45' ) x_solution = solution.y[0] v_solution = solution.y[1] print(f"ODE solved for {len(solution.t)} time points") print(f"Max displacement: {np.max(np.abs(x_solution)):.3f}") print(f"Integration successful: {solution.success}")

Large Sparse System
# Efficient sparse matrix operations from scipy import sparse from scipy.sparse import linalg # Create large sparse matrix (finite difference) n = 1000 diagonals = [np.ones(n-1), -2*np.ones(n), np.ones(n-1)] offsets = [-1, 0, 1] A = sparse.diags(diagonals, offsets, format='csr') # Right-hand side vector b = np.zeros(n) b[n//4:3*n//4] = 1.0 # Solve sparse system efficiently x = sparse.linalg.spsolve(A, b) print(f"Matrix size: {A.shape}") print(f"Non-zero elements: {A.nnz}") print(f"Sparsity: {A.nnz / (n*n) * 100:.2f}%") print(f"Solution norm: {np.linalg.norm(x):.4f}")

Robust Computing Example
# Robust scientific computing practices from scipy import optimize import warnings def robust_optimization(): # Define objective function def objective(x): if len(x) != 2: raise ValueError("Function requires 2D input") return 100.0 * (x[1] - x[0]**2)**2 + (1 - x[0])**2 # Multiple initial guesses for robustness initial_guesses = [ np.array([0.0, 0.0]), np.array([1.5, 1.5]), np.array([-1.2, 1.0]) ] best_result = None best_value = np.inf for i, x0 in enumerate(initial_guesses): try: result = optimize.minimize( objective, x0, method='BFGS', options={'disp': False} ) if result.success and result.fun < best_value: best_result = result best_value = result.fun print(f"Initial guess {i+1}: SUCCESS (f={result.fun:.2e})") else: print(f"Initial guess {i+1}: FAILED") except Exception as e: print(f"Initial guess {i+1}: ERROR - {str(e)}") if best_result: print(f"Best solution: x* = {best_result.x}") return best_result else: print("All optimization attempts failed!") return None # Run robust optimization result = robust_optimization()
