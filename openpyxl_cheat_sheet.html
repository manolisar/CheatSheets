Invoice Generation
def generate_invoice(customer_data, template_file, output_file): # Load template wb = load_workbook(template_file) ws = wb.active # Fill customer information ws['B5'] = customer_data['name'] ws['B6'] = customer_data['address'] ws['B7'] = customer_data['email'] # Add invoice items start_row = 10 total = 0 for i, item in enumerate(customer_data['items']): row = start_row + i ws[f'A{row}'] = item['description'] ws[f'B{row}'] = item['quantity'] ws[f'C{row}'] = item['price'] ws[f'D{row}'] = f'=B{row}*C{row}' # Formula for total total += item['quantity'] * item['price'] # Add total total_row = start_row + len(customer_data['items']) + 1 ws[f'C{total_row}'] = "Total:" ws[f'D{total_row}'] = total # Save personalized invoice wb.save(output_file)

Performance & Best Practices
Memory Optimization
load_workbook(read_only=True) Read-only mode for large files
Workbook(write_only=True) Write-only mode for output
ws.append(row_data) Efficient row writing
wb.close() Free memory after use
Batch Operations
ws.iter_rows(min_row=2, values_only=True) Efficient row iteration
ws.cell(row, col, value) Direct cell assignment
Performance Tip: Use write_only=True for large data exports and read_only=True for reading large files. Batch operations are faster than individual cell updates.
Memory Warning: Large Excel files can consume significant memory. Process data in chunks for files over 100MB.
Large File Processing
def process_large_file(input_file, output_file): # Read in read-only mode wb_read = load_workbook(input_file, read_only=True) ws_read = wb_read.active # Write in write-only mode wb_write = Workbook(write_only=True) ws_write = wb_write.create_sheet() # Process in batches batch_size = 1000 batch = [] for i, row in enumerate(ws_read.iter_rows(values_only=True)): # Process row data processed_row = [cell * 1.1 if isinstance(cell, (int, float)) else cell for cell in row] batch.append(processed_row) # Write batch when full if len(batch) >= batch_size: for b_row in batch: ws_write.append(b_row) batch = [] # Write remaining rows for b_row in batch: ws_write.append(b_row) # Save and close wb_write.save(output_file) wb_read.close() wb_write.close()
