Basic Programming

Program Structure

Basic Structure
void setup() {
  // Initialization code - runs once
  pinMode(ledPin, OUTPUT);
  Serial.begin(9600);
}

void loop() {
  // Main code - runs repeatedly
  digitalWrite(ledPin, HIGH);
  delay(1000);
}

Digital I/O

Pin Configuration

pinMode()
pinMode(pin, OUTPUT);        // Set pin as output
pinMode(pin, INPUT);         // Set pin as input (floating)
pinMode(pin, INPUT_PULLUP);  // Set pin as input with pullup resistor

Digital Read/Write

digitalWrite()
digitalWrite(pin, HIGH);     // Set pin to 5V (or 3.3V)
digitalWrite(pin, LOW);      // Set pin to 0V
digitalWrite(13, !digitalRead(13)); // Toggle pin state
digitalRead()
int buttonState = digitalRead(buttonPin);
if (digitalRead(2) == HIGH) {
  // Button pressed (assuming pulldown)
}

Analog I/O

Analog Input

analogRead()
int sensorValue = analogRead(A0);    // Read analog pin (0-1023)
float voltage = sensorValue * (5.0 / 1023.0); // Convert to voltage
analogReference()
analogReference(DEFAULT);    // 5V reference (or 3.3V)
analogReference(INTERNAL);   // Internal 1.1V reference
analogReference(EXTERNAL);   // External reference on AREF pin

Analog Output (PWM)

analogWrite()
analogWrite(pin, value);     // PWM output (0-255)
analogWrite(9, 128);         // 50% duty cycle on pin 9
analogWrite(6, 0);           // 0% duty cycle (OFF)
analogWrite(5, 255);         // 100% duty cycle (ON)

Serial Communication

Serial Setup

Serial.begin()
Serial.begin(9600);          // Initialize serial at 9600 baud
Serial.begin(115200);        // Initialize at 115200 baud
while (!Serial);             // Wait for serial port to connect

Serial Output

Serial Print Functions
Serial.print("Hello");       // Print without newline
Serial.println("World");     // Print with newline
Serial.print(value, DEC);    // Print decimal (default)
Serial.print(value, HEX);    // Print hexadecimal
Serial.print(value, BIN);    // Print binary
Serial.print(3.14159, 2);    // Print float with 2 decimal places

Serial Input

Serial Read Functions
if (Serial.available()) {
  char incomingByte = Serial.read();    // Read single byte
  String inputString = Serial.readString(); // Read entire string
  int number = Serial.parseInt();       // Read integer
  float decimal = Serial.parseFloat();  // Read float
}

Timing Functions

Delay Functions

Blocking Delays
delay(1000);                 // Wait 1000 milliseconds (1 second)
delayMicroseconds(50);       // Wait 50 microseconds

Non-blocking Timing

millis() and micros()
unsigned long currentTime = millis();      // Get current time in ms
unsigned long previousTime = 0;
const long interval = 1000;

if (currentTime - previousTime >= interval) {
  previousTime = currentTime;
  // Execute code every 1000ms without blocking
}

Interrupts

External Interrupts

attachInterrupt()
attachInterrupt(digitalPinToInterrupt(2), ISR_function, RISING);
// Modes: LOW, CHANGE, RISING, FALLING

void ISR_function() {
  // Interrupt service routine - keep it short!
  // No delay(), Serial.print(), or complex operations
}
detachInterrupt()
detachInterrupt(digitalPinToInterrupt(2)); // Disable interrupt

Libraries

Including Libraries

Library Management
#include <SoftwareSerial.h>  // Include library
#include "MyLibrary.h"       // Include local library

// Popular libraries:
#include <Servo.h>           // Servo motor control
#include <LiquidCrystal.h>   // LCD display
#include <Wire.h>            // I2C communication
#include <SPI.h>             // SPI communication

Common Library Examples

Servo Control
Servo myServo;
myServo.attach(9);           // Attach servo to pin 9
myServo.write(90);           // Move to 90 degrees
int pos = myServo.read();    // Read current position
LCD Display
LiquidCrystal lcd(12, 11, 5, 4, 3, 2);
lcd.begin(16, 2);            // Initialize 16x2 LCD
lcd.print("Hello, World!");  // Display text
lcd.setCursor(0, 1);         // Set cursor to column 0, line 1
lcd.clear();                 // Clear display

Board Management

Board Selection

Popular Arduino Boards

  • Arduino Mega 2560: ATmega2560, 54 digital pins, 16 analog pins

Common Issues

Troubleshooting
  • Upload Error: Check board and port selection
  • Serial Monitor Empty: Verify baud rate matches Serial.begin()
  • Code Not Running: Check for infinite loops in setup()
  • Unexpected Behavior: Use Serial.println() for debugging

Best Practices

Code Organization

Structure Guidelines
  • Use meaningful variable names: ledPin instead of p
  • Add comments explaining complex logic
  • Use constants for pin definitions
  • Avoid delay() in time-critical applications
  • Initialize all variables
  • Use functions to organize repeated code

Memory Management

Memory Tips
  • Arduino Uno has only 2KB RAM - use sparingly
  • Store large strings in program memory with F() macro
  • Avoid creating large arrays in functions
  • Use const for values that don't change
  • Consider using byte instead of int when possible

Power Management

Power Saving
  • Use sleep modes for battery-powered projects
  • Turn off unused peripherals
  • Use appropriate resistor values for LEDs
  • Consider using interrupts instead of polling
