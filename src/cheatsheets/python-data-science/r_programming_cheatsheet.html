dplyr Data Manipulation

Core Verbs

Essential Functions
library(dplyr)

# filter() - subset rows
filter(df, age > 25)
filter(df, age > 25 & height < 180)
filter(df, name %in% c("Alice", "Bob"))

# select() - subset columns
select(df, name, age)
select(df, -height)               # Exclude height
select(df, starts_with("a"))      # Columns starting with "a"
select(df, contains("age"))       # Columns containing "age"

# arrange() - sort rows
arrange(df, age)                  # Ascending
arrange(df, desc(age))            # Descending
arrange(df, name, age)            # Multiple columns

# mutate() - create/modify columns
mutate(df, 
       bmi = weight / (height/100)^2,
       age_group = ifelse(age > 30, "30+", "Under 30"))

# summarise() - aggregate data
summarise(df,
          mean_age = mean(age),
          max_height = max(height),
          count = n())

# distinct() - unique rows
distinct(df, name)                # Unique names
distinct(df)                      # Unique rows

Grouping Operations

group_by() and Operations
# Group by category
df %>%
  group_by(category) %>%
  summarise(
    mean_age = mean(age),
    count = n(),
    .groups = "drop"
  )

# Multiple grouping variables
df %>%
  group_by(category, gender) %>%
  summarise(avg_score = mean(score))

# Group and mutate
df %>%
  group_by(category) %>%
  mutate(
    avg_by_group = mean(age),
    diff_from_avg = age - avg_by_group
  )

# Window functions
df %>%
  group_by(category) %>%
  mutate(
    rank = row_number(desc(score)),
    lag_value = lag(score),
    lead_value = lead(score)
  )

tidyr Data Reshaping

Pivot Operations

Wide to Long (pivot_longer)
library(tidyr)

# Wide format data
wide_data <- data.frame(
  id = 1:3,
  jan = c(10, 15, 12),
  feb = c(11, 16, 13),
  mar = c(12, 17, 14)
)

# Convert to long format
long_data <- wide_data %>%
  pivot_longer(
    cols = jan:mar,           # Columns to pivot
    names_to = "month",       # Name for key column
    values_to = "value"       # Name for value column
  )

# More complex example
pivot_longer(
  cols = starts_with("sales_"),
  names_to = "metric",
  names_prefix = "sales_",   # Remove prefix
  values_to = "amount"
)
Long to Wide (pivot_wider)
# Long format data
long_data <- data.frame(
  id = rep(1:3, each = 3),
  month = rep(c("jan", "feb", "mar"), 3),
  value = c(10, 11, 12, 15, 16, 17, 12, 13, 14)
)

# Convert to wide format
wide_data <- long_data %>%
  pivot_wider(
    names_from = month,       # Column names come from
    values_from = value       # Values come from
  )

# Handle multiple value columns
pivot_wider(
  names_from = metric,
  values_from = c(sales, profit),
  names_sep = "_"
)

Separating and Uniting

separate() and unite()
# Separate one column into multiple
df %>%
  separate(
    col = full_name,
    into = c("first", "last"),
    sep = " "
  )

# Separate with regex
df %>%
  separate(
    col = date,
    into = c("year", "month", "day"),
    sep = "-"
  )

# Unite multiple columns into one
df %>%
  unite(
    col = "full_name",
    first, last,
    sep = " "
  )

df %>%
  unite(full_name, first, last, sep = " ", remove = TRUE)

Pattern Matching

Regex Operations
# Detect patterns
str_detect("hello world", "world")  # TRUE
str_detect(c("apple", "banana"), "a")  # c(TRUE, TRUE)

# Count matches
str_count("hello", "l")             # 2
str_count(c("apple", "banana"), "a")  # c(1, 3)

# Locate patterns
str_locate("hello world", "world")  # Start and end positions
str_locate_all("hello", "l")        # All matches

# Extract patterns
str_extract("hello123world", "\\d+")      # "123"
str_extract_all("abc123def456", "\\d+")   # List with "123", "456"

# Replace patterns
str_replace("hello world", "world", "R")     # "hello R"
str_replace_all("hello world", "l", "L")     # "heLLo worLd"

# Split strings
str_split("a,b,c", ",")             # List with vector c("a", "b", "c")
str_split("a,b,c", ",", simplify = TRUE)  # Matrix

Common Regex Patterns

Regular Expressions
# Basic patterns
"\\d"          # Any digit
"\\D"          # Any non-digit
"\\w"          # Any word character
"\\W"          # Any non-word character
"\\s"          # Any whitespace
"\\S"          # Any non-whitespace

# Quantifiers
"\\d+"         # One or more digits
"\\d*"         # Zero or more digits
"\\d?"         # Zero or one digit
"\\d{3}"       # Exactly 3 digits
"\\d{2,4}"     # 2 to 4 digits

# Anchors
"^hello"       # Starts with "hello"
"world$"       # Ends with "world"

# Character classes
"[abc]"        # Any of a, b, or c
"[a-z]"        # Any lowercase letter
"[0-9]"        # Any digit
"[^abc]"       # Anything except a, b, or c

# Examples
email_pattern <- "\\b[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\\.[A-Z|a-z]{2,}\\b"
phone_pattern <- "\\d{3}-\\d{3}-\\d{4}"

Advanced Function Concepts

Scoping and Environments
# Local vs global variables
global_var <- 10

my_function <- function() {
  local_var <- 5
  result <- global_var + local_var  # Can access global_var
  return(result)
}

# Function that modifies global variable
counter <- 0
increment <- function() {
  counter <<- counter + 1  # <<- assigns to global environment
}

# Anonymous functions
sapply(1:5, function(x) x^2)

# Functions as arguments
apply_operation <- function(x, fun) {
  fun(x)
}

apply_operation(c(1, 2, 3), sum)
apply_operation(c(1, 2, 3), mean)

# Returning functions
make_multiplier <- function(n) {
  function(x) x * n
}

double <- make_multiplier(2)
triple <- make_multiplier(3)
double(5)  # Returns 10

Apply Family

Apply Functions
# apply() - for matrices/arrays
mat <- matrix(1:12, nrow=3)
apply(mat, 1, sum)        # Sum each row (margin=1)
apply(mat, 2, mean)       # Mean of each column (margin=2)

# lapply() - returns list
numbers <- list(a=1:3, b=4:6, c=7:9)
lapply(numbers, sum)      # Sum each element, returns list

# sapply() - simplifies to vector/matrix when possible
sapply(numbers, sum)      # Returns named vector

# vapply() - like sapply but with type checking
vapply(numbers, sum, numeric(1))  # Expect single numeric result

# mapply() - multivariate version
mapply(function(x, y) x + y, 1:3, 4:6)  # Element-wise addition

# Map() - similar to mapply
Map(`+`, 1:3, 4:6)

# tapply() - apply by groups
data <- c(1, 2, 3, 4, 5, 6)
groups <- c("A", "A", "B", "B", "C", "C")
tapply(data, groups, mean)  # Mean by group

library(usethis)

# Add functions
usethis::use_r("my_function")

# Generate documentation
devtools::document()

# Check package
devtools::check()

# Install local package

# Common development commands

devtools::test()         # Run tests
usethis::use_testthat()  # Set up testing
