Performance

Performance Tips

Optimization Guidelines
# 1. Use type annotations for better performance
function fast_sum(arr::Vector{Float64})
    total = 0.0
    for x in arr
        total += x
    end
    return total
end

# 2. Avoid global variables
const GLOBAL_CONST = 10         # Constants are OK
global_var = 10                 # Avoid this in performance-critical code

function use_global()
    return global_var * 2       # Slow!
end

function use_parameter(x)
    return x * 2                # Fast!
end

# 3. Use in-place operations when possible
function update_array!(arr)
    for i in eachindex(arr)
        arr[i] *= 2
    end
end

# 4. Preallocate arrays
function slow_concat()
    result = Float64[]
    for i in 1:1000
        push!(result, i)        # Reallocates memory
    end
    return result
end

function fast_concat()
    result = Vector{Float64}(undef, 1000)  # Preallocate
    for i in 1:1000
        result[i] = i
    end
    return result
end

    # Parallel execution
    result[i] = expensive_computation(i)
end

# Distributed computing
using Distributed
addprocs(4)                     # Add 4 worker processes

@everywhere function parallel_work(x)
    return x^2
end

# Parallel map across processes
results = pmap(parallel_work, 1:100)

# Shared arrays for distributed computing
using SharedArrays
shared_arr = SharedArray{Float64}(100, 100)

@distributed for i in 1:100
    shared_arr[i, :] = rand(100)
end

# SIMD (Single Instruction, Multiple Data)
function simd_sum(arr)
    total = 0.0
    @simd for x in arr
        total += x
    end
    return total
end

Metaprogramming

Macros

Writing Macros
# Basic macro
macro sayhello(name)
    return :(println("Hello, $($name)!"))
end

@sayhello "World"               # Expands to println("Hello, World!")

# Macro with multiple arguments
macro time_it(expr)
    return quote
        start_time = time()
        result = $expr
        elapsed = time() - start_time
        println("Elapsed time: $elapsed seconds")
        result
    end
end

@time_it sum(rand(1000))

# String macros
macro r_str(pattern)
    Regex(pattern)
end

r"[0-9]+"                       # Creates a regex

# Macro expansion
@macroexpand @time_it sum(rand(1000))  # See what macro generates

# Generated functions
@generated function my_multiply(x::T, y::T) where T
    if T <: Integer
        return :(x * y)
    else
        return :(x * y + 0.0)
    end
end

Expressions & Symbols

Code as Data
# Symbols
:x                              # Symbol
:my_variable                    # Symbol with underscore
Symbol("dynamic_name")          # Create symbol from string

# Expressions
expr = :(x + y)                 # Parse expression
expr = quote                    # Multi-line expression
    x = 1
    y = 2
    x + y
end

# Expression manipulation
expr.head                       # :call
expr.args                       # [:+, :x, :y]

# Evaluate expressions
x, y = 3, 4
eval(:(x + y))                  # 7

# String interpolation in expressions
var_name = :z
expr = :($var_name = 42)        # Creates :z = 42

# Code generation
function make_function(op)
    return eval(quote
        function generated_func(x, y)
            return $op(x, y)
        end
    end)
end

add_func = make_function(:+)
add_func(3, 4)                  # 7

Useful Macros

Built-in Macros
# Performance macros
@time expression                # Time execution
@elapsed expression             # Get elapsed time
@allocated expression           # Get memory allocation

# Code introspection
@which func(args...)            # Which method is called
@code_typed func(args...)       # Type-inferred code
@code_llvm func(args...)        # LLVM IR
@code_native func(args...)      # Assembly code

# Debugging
@show variable                  # Print variable name and value
@assert condition "message"     # Assertion with message

# Array operations
@. expr                         # Broadcast entire expression
@views expr                     # Use views instead of copies

# Example with @.
a = [1, 2, 3]
b = [4, 5, 6]
@. a = sin(a) + cos(b)         # Equivalent to a .= sin.(a) .+ cos.(b)

# Example with @views
function sum_cols(matrix)
    @views sum(matrix[:, 1:end-1], dims=1)  # No copying
end

# String processing
@sprintf("Value: %.2f", 3.14159)  # Format string
